DROP TABLE PARTNER;
DROP TABLE PARTNERRANK;
DROP TABLE CAMPAIGNLEVEL1;
DROP TABLE CAMPAIGNLEVEL2;
DROP TABLE CAMPAIGNLEVEL3;
DROP TABLE CAMPAIGNLEVEL4;
DROP TABLE CAMPAIGNSTATUS;
DROP TABLE EMPLOYEE_LIST;
DROP TABLE ADMIN_RANK;
DROP TABLE CAMPAIGN;
DROP TABLE USERPASS;

CREATE TABLE CAMPAIGNLEVEL1 (
	CNO VARCHAR2 (10),
	CLEVEL1START VARCHAR2 (30),
	CLEVEL1END VARCHAR2 (30),
	PRIMARY KEY (CNO, CLEVEL1START)
);

CREATE TABLE CAMPAIGNLEVEL2 (
	CNO VARCHAR2 (10),
	CLEVEL2START VARCHAR2 (30),
	CLEVEL2END VARCHAR2 (30),
	PRIMARY KEY (CNO, CLEVEL2START)
);

CREATE TABLE CAMPAIGNLEVEL3 (
	CNO VARCHAR2 (10),
	CLEVEL3START VARCHAR2 (30),
	CLEVEL3END VARCHAR2 (30),
	PRIMARY KEY (CNO, CLEVEL3START)
);

CREATE TABLE CAMPAIGNLEVEL4 (
	CNO VARCHAR2 (10),
	CLEVEL4START VARCHAR2 (30),
	CLEVEL4END VARCHAR2 (30),
	PRIMARY KEY (CNO, CLEVEL4START)
);

CREATE TABLE CAMPAIGNSTATUS (
	CSTATUS NUMBER (10),
	CSTATUSDESCRIPTION VARCHAR2 (30),
	PRIMARY KEY (CSTATUS)
);

CREATE TABLE PARTNER (
PNO VARCHAR2(10),
PNAME VARCHAR2(80),
PCONTACTPERSON VARCHAR2(80),
PMAIL VARCHAR2(80),
PPHONE VARCHAR2(15),
PADDRESS VARCHAR2(30),
PRANK NUMBER(5),
PRIMARY KEY(PNO)
);

CREATE TABLE PARTNERRANK(
PRANK NUMBER(5),
PRANKSTATUS VARCHAR2(25),
PRIMARY KEY(PRANK)
);

CREATE TABLE EMPLOYEE_LIST(
ENO VARCHAR2(10),
ENAME VARCHAR2(80),
EMAIL VARCHAR2 (80),
EPHONE VARCHAR2 (25),
EADMINRANK NUMBER(5),
ESTATUS VARCHAR2(20),
PRIMARY KEY (ENO)
);

CREATE TABLE ADMIN_RANK(
EADMINRANK NUMBER (5),
ARANKSTATUS VARCHAR2(10),
PRIMARY KEY (EADMINRANK)
);

CREATE TABLE CAMPAIGN (
	CNO VARCHAR2 (10),
	PNAME VARCHAR2 (30),
	CCONTACT VARCHAR2 (30),
	ENAME VARCHAR2 (30),
	CSTARTDATE DATE,
	CENDDATE DATE,
	CSTATUS NUMBER(2),	
	PRIMARY KEY (CNO)
);

CREATE TABLE USERPASS (
	USERNAME VARCHAR2 (30),
	PASSWORD VARCHAR2 (30),
	PRIMARY KEY(USERNAME, PASSWORD)
);
