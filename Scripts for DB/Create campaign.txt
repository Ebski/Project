DROP TABLE CAMPAIGN;
DROP SEQUENCE SEQ_CAMPAIGN;
DROP SEQUENCE SEQ_MDF;
DROP SEQUENCE SEQ_POE;

CREATE SEQUENCE SEQ_CAMPAIGN
MINVALUE 1
START WITH 1
INCREMENT BY 1
CACHE 10;

CREATE SEQUENCE SEQ_MDF
MINVALUE 1
START WITH 1
INCREMENT BY 1
CACHE 10;

CREATE SEQUENCE SEQ_POE
MINVALUE 1
START WITH 1
INCREMENT BY 1
CACHE 10;

CREATE TABLE CAMPAIGN (
	CAMPAIGN_NO VARCHAR2 (20),
	PARTNER_NO VARCHAR2 (20),
	ID_MDF NUMBER (20),
	ID_POE NUMBER (20),
	
	PRIMARY KEY (CAMPAIGN_NO),
    FOREIGN KEY (PARTNER_NO) REFERENCES PARTNER,
    FOREIGN KEY (ID_MDF) REFERENCES MDF,
    FOREIGN KEY (ID_POE) REFERENCES POE
);
